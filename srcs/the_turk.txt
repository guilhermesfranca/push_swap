#include "../includes/push_swap.h"

static int	ft_max(int a, int b)
{
	if (a > b)
		return (a);
	return (b);
}

static int	find_target_in_b(t_stack *stackb, int value)
{
	t_stack	*current;
	int		target;

	target = stackb->value;
	current = stackb;
	while (1)
	{
		if (current->value > value && value > current->next->value)
		{
			target = current->next->index;
			break ;
		}
		if (current->next == stackb)
		{
			target = ft_stacksize(stackb) - 1;
			break ;
		}
		current = current->next;
	}
	return (target);
}

static int	calc_cost(int pos_a, int size_a, int pos_b, int size_b)
{
	int	custo_a;
	int	custo_b;

	if (pos_a <= size_a / 2)
		custo_a = pos_a;
	else
		custo_a = size_a - pos_a;
	if (pos_b <= size_b / 2)
		custo_b = pos_b;
	else
		custo_b = size_b - pos_b;
	if ((pos_a <= size_a / 2) == (pos_b <= size_b / 2))
		return (ft_max(custo_a, custo_b));
	return (custo_a + custo_b);
}

static int	find_cheapest(t_stack *a, t_stack *b)
{
	t_stack	*current;
	t_stack	*tmp;
	int		best_cost;
	int		best_index;
	int		cost;
	int		target_index;
	int		pos_a;
	int		pos_b;
	int		size_a;
	int		size_b;

	size_a = ft_stacksize(a);
	size_b = ft_stacksize(b);
	best_cost = INT_MAX;
	best_index = a->index;
	current = a;
	while (1)
	{
		pos_a = 0;
		tmp = a;
		while (tmp->index != current->index)
		{
			pos_a++;
			tmp = tmp->next;
		}
		target_index = find_target_in_b(b, current->value);
		pos_b = 0;
		tmp = b;
		while (tmp->index != target_index)
		{
			pos_b++;
			tmp = tmp->next;
		}
		cost = calc_cost(pos_a, size_a, pos_b, size_b);
		if (cost < best_cost)
		{
			best_cost = cost;
			best_index = current->index;
		}
		current = current->next;
		if (current == a)
			break ;
	}
	return (best_index);
}

static void	do_move(t_stack **a, t_stack **b, int cheapest_index)
{
	int		target_index;
	int		pos_a;
	int		pos_b;
	int		size_a;
	int		size_b;
	t_stack	*tmp;

	tmp = *a;
	while (tmp->index != cheapest_index)
		tmp = tmp->next;
	target_index = find_target_in_b(*b, tmp->value);
	pos_a = 0;
	tmp = *a;
	while (tmp->index != cheapest_index)
	{
		pos_a++;
		tmp = tmp->next;
	}
	pos_b = 0;
	tmp = *b;
	while (tmp->index != target_index)
	{
		pos_b++;
		tmp = tmp->next;
	}
	size_a = ft_stacksize(*a);
	size_b = ft_stacksize(*b);
	if (pos_a <= size_a / 2 && pos_b <= size_b / 2)
		while ((*a)->index != cheapest_index && (*b)->index != target_index)
			rr(a, b);
	else if (pos_a > size_a / 2 && pos_b > size_b / 2)
		while ((*a)->index != cheapest_index && (*b)->index != target_index)
			rrr(a, b);
	bring_to_top(a, cheapest_index, size_a);
	bring_to_top(b, target_index, size_b);
	pb(a, b);
}

static void	bring_to_top_by_value(t_stack **a, int target_value, int size)
{
	int		pos;
	t_stack	*tmp;

	pos = 0;
	tmp = *a;
	while (tmp->value != target_value)
	{
		pos++;
		tmp = tmp->next;
	}
	if (pos <= size / 2)
		while ((*a)->value != target_value)
			ra(a);
	else
		while ((*a)->value != target_value)
			rra(a);
}

static void	push_back_to_a(t_stack **a, t_stack **b)
{
	int		target_value;
	int		size_a;
	t_stack	*tmp;

	while (*b)
	{
		size_a = ft_stacksize(*a);
		tmp = *a;
		target_value = -1;
		while (1)
		{
			if (tmp->value < (*b)->value && tmp->next->value > (*b)->value)
			{
				target_value = tmp->value;
				break ;
			}
			if (tmp->next == *a)
			{
				target_value = ft_stacksize(*a) - 1;
				break ;
			}
			tmp = tmp->next;
		}
		bring_to_top_by_value(a, target_value, size_a);
		pa(b, a);
	}
}
}

void	the_turk(t_stack **a, t_stack **b)
{
	int size;
	int cheapest_index;

	size = ft_stacksize(*a);
	pb(a, b);
	pb(a, b);
	assign_indexes(*a);
	assign_indexes(*b);
	while (ft_stacksize(*a) > 3)
	{
		cheapest_index = find_cheapest(*a, *b);
		do_move(a, b, cheapest_index);
		assign_indexes(*a);
		assign_indexes(*b);
	}
	simple_3low(a);
	print_list(*a);
	print_list(*b);
	push_back_to_a(a, b);
	bring_to_top(a, 0, ft_stacksize(*a));
}
